import { Meta } from "@storybook/addon-docs/blocks";
import * as cpdLight from "@vector-im/compound-design-tokens/assets/web/js/cpdLight";

import SemanticTable from "../components/SemanticTable";

<Meta title="Tokens/Spacing" />

# Spacing

<SemanticTable
  tokens={Object.keys(cpdLight)
    .filter((name) => {
      return name.includes("Space");
    })
    .sort((a, b) => {
      return (
        parseInt(cpdLight[a].replace("px", ""), 10) -
        parseInt(cpdLight[b].replace("px", ""), 10)
      );
    })
    .map((name) => {
      return {
        preview: () => (
          <div style={{ width: cpdLight[name] }} className="cpdSpacePreview" />
        ),
        name,
        variable: cpdLight[name],
      };
    })}
/>
